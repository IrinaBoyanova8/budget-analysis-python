import pandas as pd
import matplotlib.pyplot as plt

# Leer y unir los datos
df_presupuesto = pd.read_csv('presupuesto.csv')
df_real = pd.read_csv('real.csv')
df = pd.merge(df_presupuesto, df_real, on=['Departamento', 'Mes'])
df['Desviaci√≥n (‚Ç¨)'] = df['Real (‚Ç¨)'] - df['Presupuesto (‚Ç¨)']
df['Desviaci√≥n (%)'] = (df['Desviaci√≥n (‚Ç¨)'] / df['Presupuesto (‚Ç¨)']) * 100

# Gr√°fico: desviaci√≥n promedio por departamento
df_avg = df.groupby('Departamento')[['Desviaci√≥n (‚Ç¨)', 'Desviaci√≥n (%)']].mean().reset_index()

plt.figure(figsize=(10, 6))
bars = plt.bar(df_avg['Departamento'], df_avg['Desviaci√≥n (‚Ç¨)'], color='skyblue')
plt.axhline(0, color='gray', linestyle='--')
plt.title('Desviaci√≥n promedio (‚Ç¨) por departamento')
plt.ylabel('Desviaci√≥n (‚Ç¨)')
plt.grid(axis='y', linestyle='--', alpha=0.7)

# Mostrar los valores encima de cada barra
for bar in bars:
    height = bar.get_height()
    plt.annotate(f'{height:.0f}‚Ç¨', xy=(bar.get_x() + bar.get_width() / 2, height),
                 xytext=(0, 5), textcoords='offset points', ha='center', fontsize=9)

plt.tight_layout()
plt.show()

import seaborn as sns
import matplotlib.pyplot as plt
import pandas as pd

# Leer los datos
df_presupuesto = pd.read_csv('presupuesto.csv')
df_real = pd.read_csv('real.csv')

# Unir y calcular desviaciones
df = pd.merge(df_presupuesto, df_real, on=['Departamento', 'Mes'])
df['Desviaci√≥n (‚Ç¨)'] = df['Real (‚Ç¨)'] - df['Presupuesto (‚Ç¨)']
df['Desviaci√≥n (%)'] = (df['Desviaci√≥n (‚Ç¨)'] / df['Presupuesto (‚Ç¨)']) * 100
df['Desviaci√≥n (%)'] = df['Desviaci√≥n (%)'].round(2)

# Visualizaci√≥n con seaborn
plt.figure(figsize=(12, 6))
sns.set_style("whitegrid")

# Gr√°fico de barras: desviaci√≥n por departamento y mes

    data=df,
    x='Departamento',
    y='Desviaci√≥n (‚Ç¨)',

    palette='coolwarm'
